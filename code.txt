func isprime 1
    # Cette fonction permet de déterminer si un nombre donné est premier
    j:2
    # 1 n'est pas premier par convention
    1if arg0=1
        return 0
    1endi

    # on peut tester les diviseurs jusqu'a $\sqrt{arg_0}$
    # il est inutile de tester au dela
    1while j*j<arg0+1
        tmp:arg0/j
        tmp:tmp*j
        1if tmp=arg0    # Si $\lfloor \frac{arg_0}{j} \rfloor \times j = arg_0$ 
            return 0    # alors $j$ divise $arg_0$, et comme $j > 1$, $arg_0$ n'est pas premier
        1endi
        j:j+1
    1endw
    return 1
endf

func printprimes 1

        1endi

    1endw
endf
